Example Business Questions:
•	Display all movies
SELECT * FROM movie_rating
•	Show only movie title and genre
   SELECT Title, Genre
   FROM movie_rating;
•	Find movies with ratings above 8
SELECT Title, Rating
FROM movie_rating
WHERE rating > 8;
•	Find movies released after 2015
SELECT Title, Year
FROM movie_rating
WHERE Year > 2015;
•	Find the average movie rating
SELECT AVG(Rating) AS Average_Rating
FROM movie_rating;

•	Find the highest-rated movie
SELECT Title, Rating
FROM movie_rating
ORDER BY Rating Desc
LIMIT 1;

•	Average rating by genre
SELECT Genre,
 ROUND(AVG(Rating),2) AS Avg_Rating
FROM movie_rating
GROUP BY Genre
ORDER BY Avg_Rating DESC;

•	 Total revenue generated by each genre
SELECT Genre,
 ROUND(SUM(Revenue),2) AS Total_Revenue
FROM movie_rating
GROUP BY Genre
ORDER BY Total_Revenue DESC;

•	Top 10 movies by revenue
SELECT Title, Revenue
FROM movie_rating
ORDER BY Revenue DESC
LIMIT 10;


•	Average rating by country
SELECT Country,
       ROUND(AVG(Rating),2) AS Avg_Rating
FROM movie_rating
GROUP BY Country
ORDER BY Avg_Rating DESC;

•	Movies with more than 50,000 votes
SELECT Title, Votes
FROM movie_rating
WHERE Votes > 50000
ORDER BY Votes DESC;

•	Count the number of unique countries
SELECT COUNT(DISTINCT Country) AS Total_Countries
FROM movie_rating;

•	Rank movies by revenue
SELECT Title,Revenue,
 RANK() OVER(ORDER BY Revenue DESC) AS Revenue_Rank
FROM movie_rating;

•	Top 5 genres by average revenue
SELECT Genre,
 ROUND(AVG(Revenue),2) AS Avg_Revenue
FROM movie_rating
GROUP BY Genre
ORDER BY Avg_Revenue DESC
LIMIT 5;

•	Calculate average rating for each genre
SELECT Genre,
  ROUND(AVG(Rating), 2) AS Average_Rating
FROM Movie_Rating
GROUP BY Genre
ORDER BY Average_Rating DESC;

•	Find countries with their maximum movie revenue
SELECT Country, MAX(Revenue) AS Max_Revenue
FROM movie_rating
GROUP BY Country
ORDER BY Max_Revenue DESC;

•	Find the year with the latest movie release
SELECT MAX(Year) AS Latest_Year
FROM movie_rating;

•	Find countries with an average movie rating above 5
SELECT Country, AVG(Rating) AS Avg_Rating
FROM movie_rating
GROUP BY Country
HAVING AVG(Rating) > 5;

•	Count the number of unique genres in the dataset
SELECT COUNT(DISTINCT Genre) AS Total_Genres
FROM movie_rating;

•	Find all unique countries that produced movies
SELECT DISTINCT Country
FROM movie_rating
ORDER BY Country;

# 11.	BUSINESS INSIGHTS
•	Thriller movie generated the highest revenue
-	Thriller movies contributed 18k in total revenue, making it the highest performing genre

•	Drama generated the lowest revenue among the genre
-	Drama movies generated 6k in revenue.
-	This suggests that Drama movies contributed less to total revenue compared to other genre.

•	Japan recorded the highest total rating
-	Japan has the highest movie rating among all the countries.
-	This indicated stronger audience ratings.

•	The highest movie rating is 8.90
-	The dashboard shows that the highest rated movie received a rating of 8.90.
-	It indicates the strong audience rating.

•	Thriller movie dominates the rating distribution 
-	The pie chart shows the thriller approximately 24% of total ratings.
